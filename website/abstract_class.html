abstract base class


The base class with contain pure virtual function is called abstract base. And these kind of classes never allows to create object and its only ment for inheritance.

#include <iostream>
using namespace std;
//Compiler version g++ 6.3.0
class abase{
protected:
float f1,f2;
public:
void get(){
cin>>f1>>f2;
}
virtual void sol()=0;
};
class der1: public abase
{
float rel;
public:
virtual void sol(){
rel=f1+f2;
cout<<"rel="<< rel<< endl;
}
};
class der2 : public abase
{
float rel;
public :
virtual void sol(){
rel=f1-f2;
cout<<"rel="<< rel<< endl;
}
};
int main()
{
der1 d1;
der2 d2;
cout<<"enter f2 and f2 values for additional of der1 class"<< endl;
d1.get();
d1.sol();
cout<<"enter f2 and f2 values for subtraction of der2 class"<< endl;
d2.get();
d2.sol();
return 0;
}
enter f1 and f2 values for addition of der1 class
34
46
rel=80
enter f1 and f2 values for subtraction of der2 class
67
45
rel=22
