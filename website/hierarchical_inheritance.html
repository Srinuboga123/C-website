hierarchical inheritance



Hierarchical inheritance is one kind of inheritance in which here we contain one base class and any number of derived class. Hierarchical inheritance is quite opposite to multiple inheritance.
Here by using one base class we can derive any number of derived classes.

#include < iostream>
using namespace std;
class student{
protected:
int m, p,c;
char name[10];
char course[10];
public:
void get(){
cout<<"enter the student name"<< endl;
cin>>name;
cout<<"enter the student course"<< endl;
cin>>course;
cout<<"enter the m marks"<< endl;
cin>>m;
cout<<"enter the p marks"<< endl;
cin>>p;
cout<<"enter the c marks"<< endl;
cin>>c;
}
};
class total: public student{
int sum;
public:
void total1(){
sum=m+p+c;
cout<<"the total marks of student="<< sum<< endl;
}
};
class sports : public student{
int sum;
int sportsmarks;
public:
sports(){
sportsmarks=10;
cout<<"he got 10 grace marks for participate in sports"<< endl;
}
void total(){
sum=m+p+c+sportsmarks;
cout<<"the student total marks="<< sum<< endl;
}
};
int main()
{
int i;
cout<<"press 1 if student participate in sports"<< endl;
cout<<"press 0 if student not participate in sports"<< endl;
cin>>i;
if(i==1){
sports s;
s.get();
s.total();
}
else if(i==0){
total t;
t.get();
t.total1();
}
else{
cout<<"choose the correct option"<< endl;
}
return 0;
}
