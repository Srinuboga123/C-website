single inheritance



Single inheritance is the mechanisms of deriving a one derived calss from one base class. lets see a program if the visibility mood is public

#include < iostream>
using namespace std;
class abc{
int a, b;
public:
void get(){
cout<<"enter the a and b value" << endl;
cin>>a>>b;
}
void put(){
cout<<"a=" << a<< endl;
cout<<"b="<< b<< endl;
}
};
class xyz{
int sum;
public:
void sum(){
sum=a+b;
cout<<"sum="<< sum << endl;
}
};
int main(){
xyz x;
x.get();
x.put();
x.sum();
return 0;
}
enter the a and b values
20,70
a= 20
b=70
sum= 90
